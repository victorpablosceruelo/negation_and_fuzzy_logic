73      /**
74       * Tests executing several requests in parallel.
75       */
76      @Test
77      public void testParallelRequests() throws Exception {
78          // 3.x: TestHttpConnectionManager.testGetFromMultipleThreads
79  
80          final int COUNT = 8; // adjust to execute more requests
81  
82          PoolingClientConnectionManager mgr = new PoolingClientConnectionManager();
83          mgr.setMaxTotal(COUNT/2);
84          mgr.setDefaultMaxPerRoute(COUNT/2);
85  
86          final HttpHost target = getServerHttp();
87          final HttpRoute route = new HttpRoute(target, null, false);
88          final int      rsplen = 8;
89          final String      uri = "/random/" + rsplen;
90  
91          ExecReqThread[] threads = new ExecReqThread [COUNT];
92          for (int i=0; i<COUNT; i++) {
93  
94              HttpRequest request = new BasicHttpRequest
95                  ("GET", uri, HttpVersion.HTTP_1_1);
96  
97              ExecReqThread.RequestSpec ertrs = new ExecReqThread.RequestSpec();
98              ertrs.executor = httpExecutor;
99              ertrs.processor = httpProcessor;
100             ertrs.context = new BasicHttpContext(null);
101             ertrs.params = defaultParams;
102 
103             ertrs.context.setAttribute
104                 (ExecutionContext.HTTP_TARGET_HOST, target);
105             ertrs.context.setAttribute
106                 (ExecutionContext.HTTP_REQUEST, request);
107 
108             threads[i] = new ExecReqThread(mgr, route, 5000L, ertrs);
109         }
110 
111         for (int i=0; i<threads.length; i++) {
112             threads[i].start();
113         }
114 
115         for (int i=0; i<threads.length; i++) {
116             threads[i].join(10000);
117             Assert.assertNull("exception in thread " + i,
118                        threads[i].getException());
119             Assert.assertNotNull("no response in thread " + i,
120                           threads[i].getResponse());
121             Assert.assertEquals("wrong status code in thread " + i, 200,
122                          threads[i].getResponse()
123                          .getStatusLine().getStatusCode());
124             Assert.assertNotNull("no response data in thread " + i,
125                           threads[i].getResponseData());
126             Assert.assertEquals("wrong length of data in thread" + i, rsplen,
127                          threads[i].getResponseData().length);
128         }
129 
130         mgr.shutdown();
131     }

150      * Tests releasing and re-using a connection after a response is read.
151      */
152     @Test
153     public void testReleaseConnection() throws Exception {
154 
155         PoolingClientConnectionManager mgr = new PoolingClientConnectionManager();
156         mgr.setMaxTotal(1);
157 
158         final HttpHost target = getServerHttp();
159         final HttpRoute route = new HttpRoute(target, null, false);
160         final int      rsplen = 8;
161         final String      uri = "/random/" + rsplen;
162 
163         HttpRequest request =
164             new BasicHttpRequest("GET", uri, HttpVersion.HTTP_1_1);
165 
166         ManagedClientConnection conn = getConnection(mgr, route);
167         conn.open(route, httpContext, defaultParams);
168 
169         // a new context is created for each testcase, no need to reset
170         HttpResponse response = Helper.execute(
171                 request, conn, target,
172                 httpExecutor, httpProcessor, defaultParams, httpContext);
173 
174         Assert.assertEquals("wrong status in first response",
175                      HttpStatus.SC_OK,
176                      response.getStatusLine().getStatusCode());
177         byte[] data = EntityUtils.toByteArray(response.getEntity());
178         Assert.assertEquals("wrong length of first response entity",
179                      rsplen, data.length);
180         // ignore data, but it must be read
181 
182         // check that there is no auto-release by default
183         try {
184             // this should fail quickly, connection has not been released
185             getConnection(mgr, route, 10L, TimeUnit.MILLISECONDS);
186             Assert.fail("ConnectionPoolTimeoutException should have been thrown");
187         } catch (ConnectionPoolTimeoutException e) {
188             // expected
189         }
190 
191         // release connection without marking for re-use
192         // expect the next connection obtained to be closed
193         mgr.releaseConnection(conn, -1, null);
194         conn = getConnection(mgr, route);
195         Assert.assertFalse("connection should have been closed", conn.isOpen());
196 
197         // repeat the communication, no need to prepare the request again
198         conn.open(route, httpContext, defaultParams);
199         httpContext.setAttribute(ExecutionContext.HTTP_CONNECTION, conn);
200         response = httpExecutor.execute(request, conn, httpContext);
201         httpExecutor.postProcess(response, httpProcessor, httpContext);
202 
203         Assert.assertEquals("wrong status in second response",
204                      HttpStatus.SC_OK,
205                      response.getStatusLine().getStatusCode());
206         data = EntityUtils.toByteArray(response.getEntity());
207         Assert.assertEquals("wrong length of second response entity",
208                      rsplen, data.length);
209         // ignore data, but it must be read
210 
211         // release connection after marking it for re-use
212         // expect the next connection obtained to be open
213         conn.markReusable();
214         mgr.releaseConnection(conn, -1, null);
215         conn = getConnection(mgr, route);
216         Assert.assertTrue("connection should have been open", conn.isOpen());
217 
218         // repeat the communication, no need to prepare the request again
219         httpContext.setAttribute(ExecutionContext.HTTP_CONNECTION, conn);
220         response = httpExecutor.execute(request, conn, httpContext);
221         httpExecutor.postProcess(response, httpProcessor, httpContext);
222 
223         Assert.assertEquals("wrong status in third response",
224                      HttpStatus.SC_OK,
225                      response.getStatusLine().getStatusCode());
226         data = EntityUtils.toByteArray(response.getEntity());
227         Assert.assertEquals("wrong length of third response entity",
228                      rsplen, data.length);
229         // ignore data, but it must be read
230 
231         mgr.releaseConnection(conn, -1, null);
232         mgr.shutdown();
233     }

340     @Test
341     public void testCloseExpiredIdleConnections() throws Exception {
342 
343         PoolingClientConnectionManager mgr = new PoolingClientConnectionManager();
344         mgr.setMaxTotal(1);
345 
346         final HttpHost target = getServerHttp();
347         final HttpRoute route = new HttpRoute(target, null, false);
348 
349         ManagedClientConnection conn = getConnection(mgr, route);
350         conn.open(route, httpContext, defaultParams);
351 
352         Assert.assertEquals(1, mgr.getTotalStats().getLeased());
353         Assert.assertEquals(1, mgr.getStats(route).getLeased());
354         conn.markReusable();
355         mgr.releaseConnection(conn, 100, TimeUnit.MILLISECONDS);
356 
357         // Released, still active.
358         Assert.assertEquals(1, mgr.getTotalStats().getAvailable());
359         Assert.assertEquals(1, mgr.getStats(route).getAvailable());
360 
361         mgr.closeExpiredConnections();
362 
363         // Time has not expired yet.
364         Assert.assertEquals(1, mgr.getTotalStats().getAvailable());
365         Assert.assertEquals(1, mgr.getStats(route).getAvailable());
366 
367         Thread.sleep(150);
368 
369         mgr.closeExpiredConnections();
370 
371         // Time expired now, connections are destroyed.
372         Assert.assertEquals(0, mgr.getTotalStats().getAvailable());
373         Assert.assertEquals(0, mgr.getStats(route).getAvailable());
374 
375         mgr.shutdown();
376     }

695         private final CountDownLatch continueLatch;
696         private final CountDownLatch waitLatch = new CountDownLatch(1);
697         protected final WaitPolicy waitPolicy;
698 
699         LatchSupport(CountDownLatch continueLatch, WaitPolicy waitPolicy) {
700             this.continueLatch = continueLatch;
701             this.waitPolicy = waitPolicy;
702         }
703 
704         void waitForState() throws InterruptedException {
705             if(!waitLatch.await(1, TimeUnit.SECONDS))
706                 throw new RuntimeException("waited too long");
707         }
708 
709         void latch() {
710             waitLatch.countDown();
711             try {
712                 if (!continueLatch.await(60, TimeUnit.SECONDS))
713                     throw new RuntimeException("waited too long!");
714             } catch (InterruptedException e) {
715                 throw new RuntimeException(e);
716             }
717         }
