69      @Test
70      public void testProxyAuthOverPlainConnection() throws Exception {
71          HttpRequest request = new BasicHttpRequest("GET", "/");
72          HttpContext context = new BasicHttpContext();
73  
74          HttpHost target = new HttpHost("localhost", 443, "https");
75          HttpHost proxy = new HttpHost("localhost", 8080);
76          HttpRoute route = new HttpRoute(target, null, proxy, false,
77                  TunnelType.PLAIN, LayerType.PLAIN);
78  
79          HttpRoutedConnection conn = Mockito.mock(HttpRoutedConnection.class);
80          Mockito.when(conn.getRoute()).thenReturn(route);
81  
82          BasicScheme authscheme = new BasicScheme();
83          Credentials creds = new UsernamePasswordCredentials("user", "secret");
84          BasicHeader challenge = new BasicHeader(AUTH.PROXY_AUTH, "BASIC realm=auth-realm");
85          authscheme.processChallenge(challenge);
86  
87          AuthState authstate = new AuthState();
88          authstate.update(authscheme, creds);
89  
90          context.setAttribute(ExecutionContext.HTTP_CONNECTION, conn);
91          context.setAttribute(ClientContext.PROXY_AUTH_STATE, authstate);
92  
93          HttpRequestInterceptor interceptor = new RequestProxyAuthentication();
94          interceptor.process(request, context);
95          Header header = request.getFirstHeader(AUTH.PROXY_AUTH_RESP);
96          Assert.assertNotNull(header);
97          Assert.assertEquals("Basic dXNlcjpzZWNyZXQ=", header.getValue());
98      }
