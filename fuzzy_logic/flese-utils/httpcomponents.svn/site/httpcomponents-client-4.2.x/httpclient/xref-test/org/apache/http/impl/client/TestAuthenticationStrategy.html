101     public void testGetChallenges() throws Exception {
102         TargetAuthenticationStrategy authStrategy = new TargetAuthenticationStrategy();
103         HttpContext context = new BasicHttpContext();
104         HttpHost host = new HttpHost("localhost", 80);
105         HttpResponse response = new BasicHttpResponse(HttpVersion.HTTP_1_1, HttpStatus.SC_UNAUTHORIZED, "UNAUTHORIZED");
106         Header h1 = new BasicHeader(AUTH.WWW_AUTH, "  Basic  realm=\"test\"");
107         Header h2 = new BasicHeader(AUTH.WWW_AUTH, "\t\tDigest   realm=\"realm1\", nonce=\"1234\"");
108         Header h3 = new BasicHeader(AUTH.WWW_AUTH, "WhatEver realm=\"realm1\", stuff=\"1234\"");
109         response.addHeader(h1);
110         response.addHeader(h2);
111         response.addHeader(h3);
112 
113         Map<String, Header> challenges = authStrategy.getChallenges(host, response, context);
114 
115         Assert.assertNotNull(challenges);
116         Assert.assertEquals(3, challenges.size());
117         Assert.assertSame(h1, challenges.get("basic"));
118         Assert.assertSame(h2, challenges.get("digest"));
119         Assert.assertSame(h3, challenges.get("whatever"));
120     }

250         challenges.put("whatever", new BasicHeader(AUTH.WWW_AUTH, "Whatever realm=\"realm3\""));
251 
252         AuthSchemeRegistry authSchemeRegistry = new AuthSchemeRegistry();
253         authSchemeRegistry.register("basic", new BasicSchemeFactory());
