44      static class LocalPoolEntry extends PoolEntry<String, IOSession> {
45  
46          public LocalPoolEntry(final String route, final IOSession conn) {
47              super(null, route, conn);
48          }
49  
50          @Override
51          public void close() {
52              getConnection().close();
53          }
54  
55          @Override
56          public boolean isClosed() {
57              return getConnection().isClosed();
58          }

86      @Test
87      public void testSuccessfulConnect() throws Exception {
88          LocalRoutePool pool = new LocalRoutePool();
89          IOSession session = Mockito.mock(IOSession.class);
90          SessionRequest sessionRequest = Mockito.mock(SessionRequest.class);
91          Mockito.when(sessionRequest.getSession()).thenReturn(session);
92          BasicFuture<LocalPoolEntry> future = new BasicFuture<LocalPoolEntry>(null);
93          pool.addPending(sessionRequest, future);
94          Assert.assertEquals(1, pool.getAllocatedCount());
95          Assert.assertEquals(0, pool.getAvailableCount());
96          Assert.assertEquals(0, pool.getLeasedCount());
97          Assert.assertEquals(1, pool.getPendingCount());
98          LocalPoolEntry entry = pool.createEntry(sessionRequest, session);
99          Assert.assertNotNull(entry);
100         Assert.assertSame(session, entry.getConnection());
101         Assert.assertFalse(future.isDone());

103         pool.completed(sessionRequest, entry);
104         Assert.assertTrue(future.isDone());
105         Assert.assertFalse(future.isCancelled());
106 
107         Assert.assertEquals(1, pool.getAllocatedCount());
108         Assert.assertEquals(0, pool.getAvailableCount());
109         Assert.assertEquals(1, pool.getLeasedCount());
