128     public void prepare(
129             final InternalState state,
130             final HttpHost target,
131             final HttpRequest original) throws HttpException, IOException {
132         if (this.log.isDebugEnabled()) {
133             this.log.debug("[exchange: " + state.getId() + "] start execution");
134         }
135 
136         final HttpClientContext localContext = state.getLocalContext();
137 
138         if (original instanceof Configurable) {
139             final RequestConfig config = ((Configurable) original).getConfig();
140             if (config != null) {
141                 localContext.setRequestConfig(config);
142             }
143         }
144 
145         final List<URI> redirectLocations = localContext.getRedirectLocations();
146         if (redirectLocations != null) {
147             redirectLocations.clear();
148         }
149 
150         final HttpRequestWrapper request = HttpRequestWrapper.wrap(original);
151         final HttpRoute route = this.routePlanner.determineRoute(target, request, localContext);
152         state.setRoute(route);
153         state.setMainRequest(request);
154         state.setCurrentRequest(request);
155 
156         prepareRequest(state);
157     }

159     public HttpRequest generateRequest(
160             final InternalState state,
161             final InternalConnManager connManager) throws IOException, HttpException {
162         final HttpClientContext localContext = state.getLocalContext();
163         final HttpRoute route = state.getRoute();
164         final NHttpClientConnection managedConn = connManager.getConnection();
165         if (!state.isRouteEstablished() && state.getRouteTracker() == null) {
166             state.setRouteEstablished(this.connmgr.isRouteComplete(managedConn));
167             if (!state.isRouteEstablished()) {
168                 this.log.debug("Start connection routing");
169                 state.setRouteTracker(new RouteTracker(route));
170             } else {
171                 this.log.debug("Connection route already established");
172             }
173         }
174 
175         if (!state.isRouteEstablished()) {
176             final RouteTracker routeTracker = state.getRouteTracker();
177             int step;
178             loop:
179             do {
180                 final HttpRoute fact = routeTracker.toRoute();
181                 step = this.routeDirector.nextStep(route, fact);
182                 switch (step) {
183                 case HttpRouteDirector.CONNECT_TARGET:
184                     this.connmgr.startRoute(managedConn, route, localContext);
185                     routeTracker.connectTarget(route.isSecure());
186                     break;
187                 case HttpRouteDirector.CONNECT_PROXY:
188                     this.connmgr.startRoute(managedConn, route, localContext);
189                     final HttpHost proxy  = route.getProxyHost();
190                     routeTracker.connectProxy(proxy, false);
191                     break;
192                 case HttpRouteDirector.TUNNEL_TARGET:
193                     if (this.log.isDebugEnabled()) {
194                         this.log.debug("[exchange: " + state.getId() + "] Tunnel required");
195                     }
196                     final HttpRequest connect = createConnectRequest(route, state);
197                     state.setCurrentRequest(HttpRequestWrapper.wrap(connect));
198                     break loop;
199                 case HttpRouteDirector.TUNNEL_PROXY:
200                     throw new HttpException("Proxy chains are not supported");
201                 case HttpRouteDirector.LAYER_PROTOCOL:
202                     this.connmgr.upgrade(managedConn, route, localContext);
203                     routeTracker.layerProtocol(route.isSecure());
204                     break;
205                 case HttpRouteDirector.UNREACHABLE:
206                     throw new HttpException("Unable to establish route: " +
207                             "planned = " + route + "; current = " + fact);
208                 case HttpRouteDirector.COMPLETE:
209                     this.connmgr.routeComplete(managedConn, route, localContext);
210                     state.setRouteEstablished(true);
211                     state.setRouteTracker(null);
212                     this.log.debug("Connection route established");
213                     break;
214                 default:
215                     throw new IllegalStateException("Unknown step indicator "
216                             + step + " from RouteDirector.");
217                 }
218             } while (step > HttpRouteDirector.COMPLETE);
219         }
220 
221         HttpRequestWrapper currentRequest = state.getCurrentRequest();
222         if (currentRequest == null) {
223             currentRequest = state.getMainRequest();
224             state.setCurrentRequest(currentRequest);
225         }
226 
227         if (state.isRouteEstablished()) {
228             state.incrementExecCount();
229             if (state.getExecCount() > 1) {
230                 final HttpAsyncRequestProducer requestProducer = state.getRequestProducer();
231                 if (!requestProducer.isRepeatable() && state.isRequestContentProduced()) {
232                     throw new NonRepeatableRequestException("Cannot retry request " +
233                             "with a non-repeatable request entity.");
234                 }
235                 requestProducer.resetRequest();
236             }
237             if (this.log.isDebugEnabled()) {
238                 this.log.debug("[exchange: " + state.getId() + "] Attempt " + state.getExecCount() +
239                     " to execute request");
240             }
241 
242             if (!currentRequest.containsHeader(AUTH.WWW_AUTH_RESP)) {
243                 final AuthState targetAuthState = localContext.getTargetAuthState();
244                 if (this.log.isDebugEnabled()) {
245                     this.log.debug("Target auth state: " + targetAuthState.getState());
246                 }
247                 this.authenticator.generateAuthResponse(currentRequest, targetAuthState, localContext);
248             }
249             if (!currentRequest.containsHeader(AUTH.PROXY_AUTH_RESP) && !route.isTunnelled()) {
250                 final AuthState proxyAuthState = localContext.getProxyAuthState();
251                 if (this.log.isDebugEnabled()) {
252                     this.log.debug("Proxy auth state: " + proxyAuthState.getState());
253                 }
254                 this.authenticator.generateAuthResponse(currentRequest, proxyAuthState, localContext);
255             }
256         } else {
257             if (!currentRequest.containsHeader(AUTH.PROXY_AUTH_RESP)) {
258                 final AuthState proxyAuthState = localContext.getProxyAuthState();
259                 if (this.log.isDebugEnabled()) {
260                     this.log.debug("Proxy auth state: " + proxyAuthState.getState());
261                 }
262                 this.authenticator.generateAuthResponse(currentRequest, proxyAuthState, localContext);
263             }
264         }
265 
266         localContext.setAttribute(HttpCoreContext.HTTP_CONNECTION, managedConn);
267         final RequestConfig config = localContext.getRequestConfig();
268         if (config.getSocketTimeout() > 0) {
269             managedConn.setSocketTimeout(config.getSocketTimeout());
270         }
271         return currentRequest;
272     }

274     public void produceContent(
275             final InternalState state,
276             final ContentEncoder encoder,
277             final IOControl ioctrl) throws IOException {
278         if (this.log.isDebugEnabled()) {
279             this.log.debug("[exchange: " + state.getId() + "] produce content");
280         }
281         final HttpAsyncRequestProducer requestProducer = state.getRequestProducer();
282         state.setRequestContentProduced();
283         requestProducer.produceContent(encoder, ioctrl);
284         if (encoder.isCompleted()) {
285             requestProducer.resetRequest();
286         }

298     public void responseReceived(
299             final InternalState state,
300             final HttpResponse response) throws IOException, HttpException {
301         if (this.log.isDebugEnabled()) {
302             this.log.debug("[exchange: " + state.getId() + "] Response received " + response.getStatusLine());
303         }
304         final HttpClientContext context = state.getLocalContext();
305         context.setAttribute(HttpClientContext.HTTP_RESPONSE, response);
306         this.httpProcessor.process(response, context);
307 
308         state.setCurrentResponse(response);
309 
310         if (!state.isRouteEstablished()) {
311             final int status = response.getStatusLine().getStatusCode();
312             if (status < 200) {
313                 throw new HttpException("Unexpected response to CONNECT request: " +
314                         response.getStatusLine());
315             }
316             if (status == HttpStatus.SC_OK) {
317                 final RouteTracker routeTracker = state.getRouteTracker();
318                 routeTracker.tunnelTarget(false);
319                 state.setCurrentRequest(null);
320             } else {
321                 if (!handleConnectResponse(state)) {
322                     state.setFinalResponse(response);
323                 }
324             }
325         } else {
326             if (!handleResponse(state)) {
327                 state.setFinalResponse(response);
328             }
329         }
330         if (state.getFinalResponse() != null) {
331             final HttpAsyncResponseConsumer<?> responseConsumer = state.getResponseConsumer();
332             responseConsumer.responseReceived(response);
333         }
334     }

336     public void consumeContent(
337             final InternalState state,
338             final ContentDecoder decoder,
339             final IOControl ioctrl) throws IOException {
340         if (this.log.isDebugEnabled()) {
341             this.log.debug("[exchange: " + state.getId() + "] Consume content");
342         }
343         if (state.getFinalResponse() != null) {
344             final HttpAsyncResponseConsumer<?> responseConsumer = state.getResponseConsumer();
345             responseConsumer.consumeContent(decoder, ioctrl);
346         } else {
347             final ByteBuffer tmpbuf = state.getTmpbuf();
348             tmpbuf.clear();
349             decoder.read(tmpbuf);
350         }
351     }

353     public void responseCompleted(
354             final InternalState state,
355             final InternalConnManager connManager) throws IOException, HttpException {
356         final HttpClientContext localContext = state.getLocalContext();
357         final HttpResponse currentResponse = state.getCurrentResponse();
358 
359         if (!state.isRouteEstablished()) {
360             final int status = currentResponse.getStatusLine().getStatusCode();
361             if (status == HttpStatus.SC_OK) {
362                 state.setCurrentResponse(null);
363                 return;
364             }
365         }
366 
367         if (this.connReuseStrategy.keepAlive(currentResponse, localContext)) {
368             final long validDuration = this.keepaliveStrategy.getKeepAliveDuration(
369                     currentResponse, localContext);
370             if (this.log.isDebugEnabled()) {
371                 final String s;
372                 if (validDuration > 0) {
373                     s = "for " + validDuration + " " + TimeUnit.MILLISECONDS;
374                 } else {
375                     s = "indefinitely";
376                 }
377                 this.log.debug("[exchange: " + state.getId() + "] Connection can be kept alive " + s);
378             }
379             state.setValidDuration(validDuration);
380             state.setReusable();
381         } else {
382             if (this.log.isDebugEnabled()) {
383                 this.log.debug("[exchange: " + state.getId() + "] Connection cannot be kept alive");
384             }
385             state.setNonReusable();
386             connManager.releaseConnection();
387             final AuthState proxyAuthState = localContext.getProxyAuthState();
388             if (proxyAuthState.getState() == AuthProtocolState.SUCCESS
389                     && proxyAuthState.getAuthScheme() != null
390                     && proxyAuthState.getAuthScheme().isConnectionBased()) {
391                 if (this.log.isDebugEnabled()) {
392                     this.log.debug("[exchange: " + state.getId() + "] Resetting proxy auth state");
393                 }
394                 proxyAuthState.reset();
395             }
396             final AuthState targetAuthState = localContext.getTargetAuthState();
397             if (targetAuthState.getState() == AuthProtocolState.SUCCESS
398                     && targetAuthState.getAuthScheme() != null
399                     && targetAuthState.getAuthScheme().isConnectionBased()) {
400                 if (this.log.isDebugEnabled()) {
401                     this.log.debug("[exchange: " + state.getId() + "] Resetting target auth state");
402                 }
403                 targetAuthState.reset();
404             }
405         }
406 
407         Object userToken = localContext.getUserToken();
408         if (userToken == null) {
409             userToken = this.userTokenHandler.getUserToken(localContext);
410             localContext.setAttribute(HttpClientContext.USER_TOKEN, userToken);
411         }
412 
413         if (state.getFinalResponse() != null) {
414             final HttpAsyncResponseConsumer<?> responseConsumer = state.getResponseConsumer();
415             responseConsumer.responseCompleted(localContext);
416             if (this.log.isDebugEnabled()) {
417                 this.log.debug("[exchange: " + state.getId() + "] Response processed");
418             }
419             connManager.releaseConnection();
420         } else {
421             if (state.getRedirect() != null) {
422                 final HttpUriRequest redirect = state.getRedirect();
423                 final URI uri = redirect.getURI();
424                 if (this.log.isDebugEnabled()) {
425                     this.log.debug("[exchange: " + state.getId() + "] Redirecting to '" + uri + "'");
426                 }
427                 state.setRedirect(null);
428 
429                 final HttpHost newTarget = URIUtils.extractHost(uri);
430                 if (newTarget == null) {
431                     throw new ProtocolException("Redirect URI does not specify a valid host name: " + uri);
432                 }
433 
434                 // Reset auth states if redirecting to another host
435                 final HttpRoute route = state.getRoute();
436                 if (!route.getTargetHost().equals(newTarget)) {
437                     final AuthState targetAuthState = localContext.getTargetAuthState();
438                     if (this.log.isDebugEnabled()) {
439                         this.log.debug("[exchange: " + state.getId() + "] Resetting target auth state");
440                     }
441                     targetAuthState.reset();
442                     final AuthState proxyAuthState = localContext.getProxyAuthState();
443                     final AuthScheme authScheme = proxyAuthState.getAuthScheme();
444                     if (authScheme != null && authScheme.isConnectionBased()) {
445                         if (this.log.isDebugEnabled()) {
446                             this.log.debug("[exchange: " + state.getId() + "] Resetting proxy auth state");
447                         }
448                         proxyAuthState.reset();
449                     }
450                 }
451 
452                 final HttpRequestWrapper newRequest = HttpRequestWrapper.wrap(redirect);
453                 final HttpRoute newRoute = this.routePlanner.determineRoute(
454                     newTarget, newRequest, localContext);
455                 state.setRoute(newRoute);
456                 state.setMainRequest(newRequest);
457                 state.setCurrentRequest(newRequest);
458                 if (!route.equals(newRoute)) {
459                     connManager.releaseConnection();
460                 }
461                 prepareRequest(state);
462             }
463         }
464         state.setCurrentResponse(null);
465     }

467     private void rewriteRequestURI(final InternalState state) throws ProtocolException {
468         final HttpRequestWrapper request = state.getCurrentRequest();
469         final HttpRoute route = state.getRoute();
470         try {
471             URI uri = request.getURI();
472             if (uri != null) {
473                 if (route.getProxyHost() != null && !route.isTunnelled()) {
474                     // Make sure the request URI is absolute
475                     if (!uri.isAbsolute()) {
476                         final HttpHost target = route.getTargetHost();
477                         uri = URIUtils.rewriteURI(uri, target, true);
478                     } else {
479                         uri = URIUtils.rewriteURI(uri);
480                     }
481                 } else {
482                     // Make sure the request URI is relative
483                     if (uri.isAbsolute()) {
484                         uri = URIUtils.rewriteURI(uri, null, true);
485                     } else {
486                         uri = URIUtils.rewriteURI(uri);
487                     }
488                 }
489                 request.setURI(uri);
490             }
491         } catch (final URISyntaxException ex) {
492             throw new ProtocolException("Invalid URI: " +
493                     request.getRequestLine().getUri(), ex);
494         }
495     }

497     private void prepareRequest(final InternalState state) throws IOException, HttpException {
498         final HttpClientContext localContext = state.getLocalContext();
499         final HttpRequestWrapper currentRequest = state.getCurrentRequest();
500         final HttpRoute route = state.getRoute();
501 
502         final HttpRequest original = currentRequest.getOriginal();
503         URI uri = null;
504         if (original instanceof HttpUriRequest) {
505             uri = ((HttpUriRequest) original).getURI();
506         } else {
507             final String uriString = original.getRequestLine().getUri();
508             try {
509                 uri = URI.create(uriString);
510             } catch (final IllegalArgumentException ex) {
511                 if (this.log.isDebugEnabled()) {
512                     this.log.debug("Unable to parse '" + uriString + "' as a valid URI; " +
513                         "request URI and Host header may be inconsistent", ex);
514                 }
515             }
516 
517         }
518         currentRequest.setURI(uri);
519 
520         // Re-write request URI if needed
521         rewriteRequestURI(state);
522 
523         HttpHost target = null;
524         if (uri != null && uri.isAbsolute() && uri.getHost() != null) {
525             target = new HttpHost(uri.getHost(), uri.getPort(), uri.getScheme());
526         }
527         if (target == null) {
528             target = route.getTargetHost();
529         }
530 
531         // Get user info from the URI
532         if (uri != null) {
533             final String userinfo = uri.getUserInfo();
534             if (userinfo != null) {
535                 final CredentialsProvider credsProvider = localContext.getCredentialsProvider();
536                 credsProvider.setCredentials(
537                         new AuthScope(target),
538                         new UsernamePasswordCredentials(userinfo));
539             }
540         }
541 
542         localContext.setAttribute(HttpClientContext.HTTP_REQUEST, currentRequest);
543         localContext.setAttribute(HttpClientContext.HTTP_TARGET_HOST, target);
544         localContext.setAttribute(HttpClientContext.HTTP_ROUTE, route);
545         this.httpProcessor.process(currentRequest, localContext);
546     }

548     private HttpRequest createConnectRequest(
549             final HttpRoute route, final InternalState state) throws IOException, HttpException {
550         // see RFC 2817, section 5.2 and
551         // INTERNET-DRAFT: Tunneling TCP based protocols through
552         // Web proxy servers
553         final HttpHost target = route.getTargetHost();
554         final String host = target.getHostName();
555         final int port = target.getPort();
556         final StringBuilder buffer = new StringBuilder(host.length() + 6);
557         buffer.append(host);
558         buffer.append(':');
559         buffer.append(Integer.toString(port));
560         final HttpRequest request = new BasicHttpRequest("CONNECT", buffer.toString(), HttpVersion.HTTP_1_1);
561         final HttpClientContext localContext = state.getLocalContext();
562         this.proxyHttpProcessor.process(request, localContext);
563         return request;
564     }

566     private boolean handleConnectResponse(final InternalState state) throws HttpException {
567         final HttpClientContext localContext = state.getLocalContext();
568         final RequestConfig config = localContext.getRequestConfig();
569         if (config.isAuthenticationEnabled()) {
570             final CredentialsProvider credsProvider = localContext.getCredentialsProvider();
571             if (credsProvider != null) {
572                 final HttpRoute route = state.getRoute();
573                 final HttpHost proxy = route.getProxyHost();
574                 final HttpResponse currentResponse = state.getCurrentResponse();
575                 final AuthState proxyAuthState = localContext.getProxyAuthState();
576                 if (this.authenticator.isAuthenticationRequested(proxy, currentResponse,
577                         this.proxyAuthStrategy, proxyAuthState, localContext)) {
578                     return this.authenticator.handleAuthChallenge(proxy, currentResponse,
579                             this.proxyAuthStrategy, proxyAuthState, localContext);
580                 }
581             }
582         }
583         return false;
584     }

586     private boolean handleResponse(final InternalState state) throws HttpException {
587         final HttpClientContext localContext = state.getLocalContext();
588         final RequestConfig config = localContext.getRequestConfig();
589         if (config.isAuthenticationEnabled()) {
590             if (needAuthentication(state)) {
591                 // discard previous auth headers
592                 final HttpRequest currentRequest = state.getCurrentRequest();
593                 currentRequest.removeHeaders(AUTH.WWW_AUTH_RESP);
594                 currentRequest.removeHeaders(AUTH.PROXY_AUTH_RESP);
595                 return true;
596             }
597         }
598         if (config.isRedirectsEnabled()) {
599             final HttpRequest currentRequest = state.getCurrentRequest();
600             final HttpResponse currentResponse = state.getCurrentResponse();
601             if (this.redirectStrategy.isRedirected(currentRequest, currentResponse, localContext)) {
602                 final int maxRedirects = config.getMaxRedirects() >= 0 ? config.getMaxRedirects() : 100;
603                 if (state.getRedirectCount() >= maxRedirects) {
604                     throw new RedirectException("Maximum redirects (" + maxRedirects + ") exceeded");
605                 }
606                 state.incrementRedirectCount();
607                 final HttpUriRequest redirect = this.redirectStrategy.getRedirect(currentRequest, currentResponse,
608                     localContext);
609                 state.setRedirect(redirect);
610                 return true;
611             }
612         }
613         return false;
614     }

616     private boolean needAuthentication(final InternalState state) throws HttpException {
617         final HttpClientContext localContext = state.getLocalContext();
618         final CredentialsProvider credsProvider = localContext.getCredentialsProvider();
619         if (credsProvider != null) {
620             final HttpRoute route = state.getRoute();
621             final HttpResponse currentResponse = state.getCurrentResponse();
622             HttpHost target = localContext.getTargetHost();
623             if (target == null) {
624                 target = route.getTargetHost();
625             }
626             if (target.getPort() < 0) {
627                 target = new HttpHost(
628                         target.getHostName(),
629                         route.getTargetHost().getPort(),
630                         target.getSchemeName());
631             }
632             final AuthState targetAuthState = localContext.getTargetAuthState();
633             if (this.authenticator.isAuthenticationRequested(target, currentResponse,
