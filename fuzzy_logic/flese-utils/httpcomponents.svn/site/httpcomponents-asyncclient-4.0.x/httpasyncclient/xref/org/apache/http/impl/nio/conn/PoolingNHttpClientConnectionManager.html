72  /**
73   * <tt>PoolingNHttpClientConnectionManager</tt> maintains a pool of
74   * {@link NHttpClientConnection}s and is able to service connection requests
75   * from multiple execution threads. Connections are pooled on a per route
76   * basis. A request for a route which already the manager has persistent
77   * connections for available in the pool will be services by leasing
78   * a connection from the pool rather than creating a brand new connection.
79   * <p/>
80   * <tt>PoolingNHttpClientConnectionManager</tt> maintains a maximum limit
81   * of connection on a per route basis and in total. Per default this
82   * implementation will create no more than than 2 concurrent connections
83   * per given route and no more 20 connections in total. For many real-world
84   * applications these limits may prove too constraining, especially if they
85   * use HTTP as a transport protocol for their services. Connection limits,
86   * however, can be adjusted using {@link ConnPoolControl} methods.
87   *
88   * @since 4.0
89   */
90  @ThreadSafe
91  public class PoolingNHttpClientConnectionManager
92         implements NHttpClientConnectionManager, ConnPoolControl<HttpRoute> {

213     private String formatStats(final HttpRoute route) {
214         final StringBuilder buf = new StringBuilder();
215         final PoolStats totals = this.pool.getTotalStats();
216         final PoolStats stats = this.pool.getStats(route);
217         buf.append("[total kept alive: ").append(totals.getAvailable()).append("; ");
218         buf.append("route allocated: ").append(stats.getLeased() + stats.getAvailable());
219         buf.append(" of ").append(stats.getMax()).append("; ");
220         buf.append("total allocated: ").append(totals.getLeased() + totals.getAvailable());
221         buf.append(" of ").append(totals.getMax()).append("]");
222         return buf.toString();
223     }

236     public Future<NHttpClientConnection> requestConnection(
237             final HttpRoute route,
238             final Object state,
239             final long connectTimeout,
240             final long leaseTimeout,
241             final TimeUnit tunit,
242             final FutureCallback<NHttpClientConnection> callback) {
243         Args.notNull(route, "HTTP route");
244         if (this.log.isDebugEnabled()) {
245             this.log.debug("Connection request: " + format(route, state) + formatStats(route));
246         }
247         final BasicFuture<NHttpClientConnection> future = new BasicFuture<NHttpClientConnection>(callback);
248         final HttpHost host;
249         if (route.getProxyHost() != null) {
250             host = route.getProxyHost();
251         } else {
252             host = route.getTargetHost();
253         }
254         final SchemeIOSessionStrategy sf = this.iosessionFactoryRegistry.lookup(
255                 host.getSchemeName());
256         if (sf == null) {
257             future.failed(new UnsupportedSchemeException(host.getSchemeName() +
258                     " protocol is not supported"));
259             return future;
260         }
261         this.pool.lease(route, state,
262                 connectTimeout, leaseTimeout, tunit != null ? tunit : TimeUnit.MILLISECONDS,
263                 new InternalPoolEntryCallback(future));
264         return future;
265     }

267     public void releaseConnection(
268             final NHttpClientConnection managedConn,
269             final Object state,
270             final long keepalive,
271             final TimeUnit tunit) {
272         Args.notNull(managedConn, "Managed connection");
273         synchronized (managedConn) {
274             final CPoolEntry entry = CPoolProxy.detach(managedConn);
275             if (entry == null) {
276                 return;
277             }
278             final NHttpClientConnection conn = entry.getConnection();
279             try {
280                 if (conn.isOpen()) {
281                     entry.setState(state);
282                     entry.updateExpiry(keepalive, tunit != null ? tunit : TimeUnit.MILLISECONDS);
283                     if (this.log.isDebugEnabled()) {
284                         final String s;
285                         if (keepalive > 0) {
286                             s = "for " + (double) keepalive / 1000 + " seconds";
287                         } else {
288                             s = "indefinitely";
289                         }
290                         this.log.debug("Connection " + format(entry) + " can be kept alive " + s);
291                     }
292                 }
293             } finally {
294                 this.pool.release(entry, conn.isOpen() && entry.isRouteComplete());
295                 if (this.log.isDebugEnabled()) {
296                     this.log.debug("Connection released: " + format(entry) + formatStats(entry.getRoute()));
297                 }
298             }
299         }
300     }

312     public void startRoute(
313             final NHttpClientConnection managedConn,
314             final HttpRoute route,
315             final HttpContext context) throws IOException {
316         Args.notNull(managedConn, "Managed connection");
317         Args.notNull(route, "HTTP route");
318         final HttpHost host;
319         if (route.getProxyHost() != null) {
320             host = route.getProxyHost();
321         } else {
322             host = route.getTargetHost();
323         }
324         final Lookup<SchemeIOSessionStrategy> reg = getIOSessionFactoryRegistry(context);
325         final SchemeIOSessionStrategy sf = reg.lookup(host.getSchemeName());
326         if (sf == null) {
327             throw new UnsupportedSchemeException(host.getSchemeName() +
328                     " protocol is not supported");
329         }
330         if (sf.isLayeringRequired()) {
331             synchronized (managedConn) {
332                 final CPoolEntry entry = CPoolProxy.getPoolEntry(managedConn);
333                 final ManagedNHttpClientConnection conn = entry.getConnection();
334                 final IOSession currentSession = sf.upgrade(host, conn.getIOSession());
335                 conn.bind(currentSession);
336             }
337         }
338     }

340     public void upgrade(
341             final NHttpClientConnection managedConn,
342             final HttpRoute route,
343             final HttpContext context) throws IOException {
344         Args.notNull(managedConn, "Managed connection");
345         Args.notNull(route, "HTTP route");
346         final HttpHost host  = route.getTargetHost();
347         final Lookup<SchemeIOSessionStrategy> reg = getIOSessionFactoryRegistry(context);
348         final SchemeIOSessionStrategy sf = reg.lookup(host.getSchemeName());
349         if (sf == null) {
350             throw new UnsupportedSchemeException(host.getSchemeName() +
351                     " protocol is not supported");
352         }
353         if (!sf.isLayeringRequired()) {
354             throw new UnsupportedSchemeException(host.getSchemeName() +
355                     " protocol does not support connection upgrade");
356         }
357         synchronized (managedConn) {
358             final CPoolEntry entry = CPoolProxy.getPoolEntry(managedConn);
359             final ManagedNHttpClientConnection conn = entry.getConnection();
360             final IOSession currentSession = sf.upgrade(host, conn.getIOSession());
361             conn.bind(currentSession);
362         }
363     }

446     class InternalPoolEntryCallback implements FutureCallback<CPoolEntry> {
447 
448         private final BasicFuture<NHttpClientConnection> future;
449 
450         public InternalPoolEntryCallback(
451                 final BasicFuture<NHttpClientConnection> future) {
452             super();
453             this.future = future;
454         }
455 
456         public void completed(final CPoolEntry entry) {
457             Asserts.check(entry.getConnection() != null, "Pool entry with no connection");
458             if (log.isDebugEnabled()) {
459                 log.debug("Connection leased: " + format(entry) + formatStats(entry.getRoute()));
460             }
461             final NHttpClientConnection managedConn = CPoolProxy.newProxy(entry);
462             if (!this.future.completed(managedConn)) {
463                 pool.release(entry, true);
464             }
465         }
466 
467         public void failed(final Exception ex) {
468             if (log.isDebugEnabled()) {
469                 log.debug("Connection request failed", ex);
470             }
471             this.future.failed(ex);
472         }
473 
474         public void cancelled() {
475             log.debug("Connection request cancelled");
476             this.future.cancel(true);
477         }

509     static class InternalConnectionFactory implements NIOConnFactory<HttpRoute, ManagedNHttpClientConnection> {
510 
511         private final ConfigData configData;
512         private final NHttpConnectionFactory<ManagedNHttpClientConnection> connFactory;
513 
514         InternalConnectionFactory(
515                 final ConfigData configData,
516                 final NHttpConnectionFactory<ManagedNHttpClientConnection> connFactory) {
517             super();
518             this.configData = configData != null ? configData : new ConfigData();
519             this.connFactory = connFactory != null ? connFactory :
520                 ManagedNHttpClientConnectionFactory.INSTANCE;
521         }
522 
523         public ManagedNHttpClientConnection create(
524                 final HttpRoute route, final IOSession iosession) throws IOException {
525             ConnectionConfig config = null;
526             if (route.getProxyHost() != null) {
527                 config = this.configData.getConnectionConfig(route.getProxyHost());
528             }
529             if (config == null) {
530                 config = this.configData.getConnectionConfig(route.getTargetHost());
531             }
532             if (config == null) {
533                 config = this.configData.getDefaultConnectionConfig();
534             }
535             if (config == null) {
536                 config = ConnectionConfig.DEFAULT;
537             }
538             final ManagedNHttpClientConnection conn = this.connFactory.create(iosession, config);
539             iosession.setAttribute(IOEventDispatch.CONNECTION_KEY, conn);
540             return conn;
541         }

545     static class InternalAddressResolver implements SocketAddressResolver<HttpRoute> {
546 
547         private final SchemePortResolver schemePortResolver;
548         private final DnsResolver dnsResolver;
549 
550         public InternalAddressResolver(
551                 final SchemePortResolver schemePortResolver,
552                 final DnsResolver dnsResolver) {
553             super();
554             this.schemePortResolver = schemePortResolver != null ? schemePortResolver :
555                 DefaultSchemePortResolver.INSTANCE;
556             this.dnsResolver = dnsResolver != null ? dnsResolver :
557                     SystemDefaultDnsResolver.INSTANCE;
558         }
559 
560         public SocketAddress resolveLocalAddress(final HttpRoute route) throws IOException {
561             return route.getLocalAddress() != null ? new InetSocketAddress(route.getLocalAddress(), 0) : null;
562         }
563 
564         public SocketAddress resolveRemoteAddress(final HttpRoute route) throws IOException {
565             final HttpHost host;
566             if (route.getProxyHost() != null) {
567                 host = route.getProxyHost();
568             } else {
569                 host = route.getTargetHost();
570             }
571             final int port = this.schemePortResolver.resolve(host);
572             final InetAddress[] addresses = this.dnsResolver.resolve(host.getHostName());
573             return new InetSocketAddress(addresses[0], port);
574         }
