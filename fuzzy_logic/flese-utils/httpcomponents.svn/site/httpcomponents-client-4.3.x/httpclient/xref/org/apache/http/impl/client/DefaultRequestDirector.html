324     private RequestWrapper wrapRequest(
325             final HttpRequest request) throws ProtocolException {
326         if (request instanceof HttpEntityEnclosingRequest) {
327             return new EntityEnclosingRequestWrapper(
328                     (HttpEntityEnclosingRequest) request);
329         } else {
330             return new RequestWrapper(
331                     request);
332         }
333     }

336     protected void rewriteRequestURI(
337             final RequestWrapper request,
338             final HttpRoute route) throws ProtocolException {
339         try {
340 
341             URI uri = request.getURI();
342             if (route.getProxyHost() != null && !route.isTunnelled()) {
343                 // Make sure the request URI is absolute
344                 if (!uri.isAbsolute()) {
345                     final HttpHost target = route.getTargetHost();
346                     uri = URIUtils.rewriteURI(uri, target, true);
347                 } else {
348                     uri = URIUtils.rewriteURI(uri);
349                 }
350             } else {
351                 // Make sure the request URI is relative
352                 if (uri.isAbsolute()) {
353                     uri = URIUtils.rewriteURI(uri, null, true);
354                 } else {
355                     uri = URIUtils.rewriteURI(uri);
356                 }
357             }
358             request.setURI(uri);
359 
360         } catch (final URISyntaxException ex) {
361             throw new ProtocolException("Invalid URI: " +
362                     request.getRequestLine().getUri(), ex);
363         }
364     }

367     // non-javadoc, see interface ClientRequestDirector
368     public HttpResponse execute(final HttpHost targetHost, final HttpRequest request,
369                                 final HttpContext context)
370         throws HttpException, IOException {
371 
372         context.setAttribute(ClientContext.TARGET_AUTH_STATE, targetAuthState);
373         context.setAttribute(ClientContext.PROXY_AUTH_STATE, proxyAuthState);
374 
375         HttpHost target = targetHost;
376 
377         final HttpRequest orig = request;
378         final RequestWrapper origWrapper = wrapRequest(orig);
379         origWrapper.setParams(params);
380         final HttpRoute origRoute = determineRoute(target, origWrapper, context);
381 
382         virtualHost = (HttpHost) origWrapper.getParams().getParameter(ClientPNames.VIRTUAL_HOST);
383 
384         // HTTPCLIENT-1092 - add the port if necessary
385         if (virtualHost != null && virtualHost.getPort() == -1) {
386             final HttpHost host = (target != null) ? target : origRoute.getTargetHost();
387             final int port = host.getPort();
388             if (port != -1){
389                 virtualHost = new HttpHost(virtualHost.getHostName(), port, virtualHost.getSchemeName());
390             }
391         }
392 
393         RoutedRequest roureq = new RoutedRequest(origWrapper, origRoute);
394 
395         boolean reuse = false;
396         boolean done = false;
397         try {
398             HttpResponse response = null;
399             while (!done) {
400                 // In this loop, the RoutedRequest may be replaced by a
401                 // followup request and route. The request and route passed
402                 // in the method arguments will be replaced. The original
403                 // request is still available in 'orig'.
404 
405                 final RequestWrapper wrapper = roureq.getRequest();
406                 final HttpRoute route = roureq.getRoute();
407                 response = null;
408 
409                 // See if we have a user token bound to the execution context
410                 Object userToken = context.getAttribute(ClientContext.USER_TOKEN);
411 
412                 // Allocate connection if needed
413                 if (managedConn == null) {
414                     final ClientConnectionRequest connRequest = connManager.requestConnection(
415                             route, userToken);
416                     if (orig instanceof AbortableHttpRequest) {
417                         ((AbortableHttpRequest) orig).setConnectionRequest(connRequest);
418                     }
419 
420                     final long timeout = HttpClientParams.getConnectionManagerTimeout(params);
421                     try {
422                         managedConn = connRequest.getConnection(timeout, TimeUnit.MILLISECONDS);
423                     } catch(final InterruptedException interrupted) {
424                         Thread.currentThread().interrupt();
425                         throw new InterruptedIOException();
426                     }
427 
428                     if (HttpConnectionParams.isStaleCheckingEnabled(params)) {
429                         // validate connection
430                         if (managedConn.isOpen()) {
431                             this.log.debug("Stale connection check");
432                             if (managedConn.isStale()) {
433                                 this.log.debug("Stale connection detected");
434                                 managedConn.close();
435                             }
436                         }
437                     }
438                 }
439 
440                 if (orig instanceof AbortableHttpRequest) {
441                     ((AbortableHttpRequest) orig).setReleaseTrigger(managedConn);
442                 }
443 
444                 try {
445                     tryConnect(roureq, context);
446                 } catch (final TunnelRefusedException ex) {
447                     if (this.log.isDebugEnabled()) {
448                         this.log.debug(ex.getMessage());
449                     }
450                     response = ex.getResponse();
451                     break;
452                 }
453 
454                 final String userinfo = wrapper.getURI().getUserInfo();
455                 if (userinfo != null) {
456                     targetAuthState.update(
457                             new BasicScheme(), new UsernamePasswordCredentials(userinfo));
458                 }
459 
460                 // Get target.  Even if there's virtual host, we may need the target to set the port.
461                 if (virtualHost != null) {
462                     target = virtualHost;
463                 } else {
464                     final URI requestURI = wrapper.getURI();
465                     if (requestURI.isAbsolute()) {
466                         target = URIUtils.extractHost(requestURI);
467                     }
468                 }
469                 if (target == null) {
470                     target = route.getTargetHost();
471                 }
472 
473                 // Reset headers on the request wrapper
474                 wrapper.resetHeaders();
475                 // Re-write request URI if needed
476                 rewriteRequestURI(wrapper, route);
477 
478                 // Populate the execution context
479                 context.setAttribute(ExecutionContext.HTTP_TARGET_HOST, target);
480                 context.setAttribute(ClientContext.ROUTE, route);
481                 context.setAttribute(ExecutionContext.HTTP_CONNECTION, managedConn);
482 
483                 // Run request protocol interceptors
484                 requestExec.preProcess(wrapper, httpProcessor, context);
485 
486                 response = tryExecute(roureq, context);
487                 if (response == null) {
488                     // Need to start over
489                     continue;
490                 }
491 
492                 // Run response protocol interceptors
493                 response.setParams(params);
494                 requestExec.postProcess(response, httpProcessor, context);
495 
496 
497                 // The connection is in or can be brought to a re-usable state.
498                 reuse = reuseStrategy.keepAlive(response, context);
499                 if (reuse) {
500                     // Set the idle duration of this connection
501                     final long duration = keepAliveStrategy.getKeepAliveDuration(response, context);
502                     if (this.log.isDebugEnabled()) {
503                         final String s;
504                         if (duration > 0) {
505                             s = "for " + duration + " " + TimeUnit.MILLISECONDS;
506                         } else {
507                             s = "indefinitely";
508                         }
509                         this.log.debug("Connection can be kept alive " + s);
510                     }
511                     managedConn.setIdleDuration(duration, TimeUnit.MILLISECONDS);
512                 }
513 
514                 final RoutedRequest followup = handleResponse(roureq, response, context);
515                 if (followup == null) {
516                     done = true;
517                 } else {
518                     if (reuse) {
519                         // Make sure the response body is fully consumed, if present
520                         final HttpEntity entity = response.getEntity();
521                         EntityUtils.consume(entity);
522                         // entity consumed above is not an auto-release entity,
523                         // need to mark the connection re-usable explicitly
524                         managedConn.markReusable();
525                     } else {
526                         managedConn.close();
527                         if (proxyAuthState.getState().compareTo(AuthProtocolState.CHALLENGED) > 0
528                                 && proxyAuthState.getAuthScheme() != null
529                                 && proxyAuthState.getAuthScheme().isConnectionBased()) {
530                             this.log.debug("Resetting proxy auth state");
531                             proxyAuthState.reset();
532                         }
533                         if (targetAuthState.getState().compareTo(AuthProtocolState.CHALLENGED) > 0
534                                 && targetAuthState.getAuthScheme() != null
535                                 && targetAuthState.getAuthScheme().isConnectionBased()) {
536                             this.log.debug("Resetting target auth state");
537                             targetAuthState.reset();
538                         }
539                     }
540                     // check if we can use the same connection for the followup
541                     if (!followup.getRoute().equals(roureq.getRoute())) {
542                         releaseConnection();
543                     }
544                     roureq = followup;
545                 }
546 
547                 if (managedConn != null) {
548                     if (userToken == null) {
549                         userToken = userTokenHandler.getUserToken(context);
550                         context.setAttribute(ClientContext.USER_TOKEN, userToken);
551                     }
552                     if (userToken != null) {
553                         managedConn.setState(userToken);
554                     }
555                 }
556 
557             } // while not done
558 
559 
560             // check for entity, release connection if possible
561             if ((response == null) || (response.getEntity() == null) ||
562                 !response.getEntity().isStreaming()) {
563                 // connection not needed and (assumed to be) in re-usable state
564                 if (reuse) {
565                     managedConn.markReusable();
566                 }
567                 releaseConnection();
568             } else {
569                 // install an auto-release entity
570                 HttpEntity entity = response.getEntity();
571                 entity = new BasicManagedEntity(entity, managedConn, reuse);
572                 response.setEntity(entity);
573             }
574 
575             return response;
576 
577         } catch (final ConnectionShutdownException ex) {
578             final InterruptedIOException ioex = new InterruptedIOException(
579                     "Connection has been shut down");
580             ioex.initCause(ex);
581             throw ioex;
582         } catch (final HttpException ex) {
583             abortConnection();
584             throw ex;
585         } catch (final IOException ex) {
586             abortConnection();
587             throw ex;
588         } catch (final RuntimeException ex) {
589             abortConnection();
590             throw ex;
591         }
592     } // execute

594     /**
595      * Establish connection either directly or through a tunnel and retry in case of
596      * a recoverable I/O failure
597      */
598     private void tryConnect(
599             final RoutedRequest req, final HttpContext context) throws HttpException, IOException {
600         final HttpRoute route = req.getRoute();
601         final HttpRequest wrapper = req.getRequest();
602 
603         int connectCount = 0;
604         for (;;) {
605             context.setAttribute(ExecutionContext.HTTP_REQUEST, wrapper);
606             // Increment connect count
607             connectCount++;
608             try {
609                 if (!managedConn.isOpen()) {
610                     managedConn.open(route, context, params);
611                 } else {
612                     managedConn.setSocketTimeout(HttpConnectionParams.getSoTimeout(params));
613                 }
614                 establishRoute(route, context);
615                 break;
616             } catch (final IOException ex) {
617                 try {
618                     managedConn.close();
619                 } catch (final IOException ignore) {
620                 }
621                 if (retryHandler.retryRequest(ex, connectCount, context)) {
622                     if (this.log.isInfoEnabled()) {
623                         this.log.info("I/O exception ("+ ex.getClass().getName() +
624                                 ") caught when connecting to the target host: "
625                                 + ex.getMessage());
626                         if (this.log.isDebugEnabled()) {
627                             this.log.debug(ex.getMessage(), ex);
628                         }
629                         this.log.info("Retrying connect");
630                     }
631                 } else {
632                     throw ex;
633                 }
634             }
635         }
636     }

638     /**
639      * Execute request and retry in case of a recoverable I/O failure
640      */
641     private HttpResponse tryExecute(
642             final RoutedRequest req, final HttpContext context) throws HttpException, IOException {
643         final RequestWrapper wrapper = req.getRequest();
644         final HttpRoute route = req.getRoute();
645         HttpResponse response = null;
646 
647         Exception retryReason = null;
648         for (;;) {
649             // Increment total exec count (with redirects)
650             execCount++;
651             // Increment exec count for this particular request
652             wrapper.incrementExecCount();
653             if (!wrapper.isRepeatable()) {
654                 this.log.debug("Cannot retry non-repeatable request");
655                 if (retryReason != null) {
656                     throw new NonRepeatableRequestException("Cannot retry request " +
657                         "with a non-repeatable request entity.  The cause lists the " +
658                         "reason the original request failed.", retryReason);
659                 } else {
660                     throw new NonRepeatableRequestException("Cannot retry request " +
661                             "with a non-repeatable request entity.");
662                 }
663             }
664 
665             try {
666                 if (!managedConn.isOpen()) {
667                     // If we have a direct route to the target host
668                     // just re-open connection and re-try the request
669                     if (!route.isTunnelled()) {
670                         this.log.debug("Reopening the direct connection.");
671                         managedConn.open(route, context, params);
672                     } else {
673                         // otherwise give up
674                         this.log.debug("Proxied connection. Need to start over.");
675                         break;
676                     }
677                 }
678 
679                 if (this.log.isDebugEnabled()) {
680                     this.log.debug("Attempt " + execCount + " to execute request");
681                 }
682                 response = requestExec.execute(wrapper, managedConn, context);
683                 break;
684 
685             } catch (final IOException ex) {
686                 this.log.debug("Closing the connection.");
687                 try {
688                     managedConn.close();
689                 } catch (final IOException ignore) {
690                 }
691                 if (retryHandler.retryRequest(ex, wrapper.getExecCount(), context)) {
692                     if (this.log.isInfoEnabled()) {
693                         this.log.info("I/O exception ("+ ex.getClass().getName() +
694                                 ") caught when processing request: "
695                                 + ex.getMessage());
696                     }
697                     if (this.log.isDebugEnabled()) {
698                         this.log.debug(ex.getMessage(), ex);
699                     }
700                     this.log.info("Retrying request");
701                     retryReason = ex;
702                 } else {
703                     throw ex;
704                 }
705             }
706         }
707         return response;
708     }

755     /**
756      * Establishes the target route.
757      *
758      * @param route     the route to establish
759      * @param context   the context for the request execution
760      *
761      * @throws HttpException    in case of a problem
762      * @throws IOException      in case of an IO problem
763      */
764     protected void establishRoute(final HttpRoute route, final HttpContext context)
765         throws HttpException, IOException {
766 
767         final HttpRouteDirector rowdy = new BasicRouteDirector();
768         int step;
769         do {
770             final HttpRoute fact = managedConn.getRoute();
771             step = rowdy.nextStep(route, fact);
772 
773             switch (step) {
774 
775             case HttpRouteDirector.CONNECT_TARGET:
776             case HttpRouteDirector.CONNECT_PROXY:
777                 managedConn.open(route, context, this.params);
778                 break;
779 
780             case HttpRouteDirector.TUNNEL_TARGET: {
781                 final boolean secure = createTunnelToTarget(route, context);
782                 this.log.debug("Tunnel to target created.");
783                 managedConn.tunnelTarget(secure, this.params);
784             }   break;
785 
786             case HttpRouteDirector.TUNNEL_PROXY: {
787                 // The most simple example for this case is a proxy chain
788                 // of two proxies, where P1 must be tunnelled to P2.
789                 // route: Source -> P1 -> P2 -> Target (3 hops)
790                 // fact:  Source -> P1 -> Target       (2 hops)
791                 final int hop = fact.getHopCount()-1; // the hop to establish
792                 final boolean secure = createTunnelToProxy(route, hop, context);
793                 this.log.debug("Tunnel to proxy created.");
794                 managedConn.tunnelProxy(route.getHopTarget(hop),
795                                         secure, this.params);
796             }   break;
797 
798 
799             case HttpRouteDirector.LAYER_PROTOCOL:
800                 managedConn.layerProtocol(context, this.params);
801                 break;
802 
803             case HttpRouteDirector.UNREACHABLE:
804                 throw new HttpException("Unable to establish route: " +
805                         "planned = " + route + "; current = " + fact);
806             case HttpRouteDirector.COMPLETE:
807                 // do nothing
808                 break;
809             default:
810                 throw new IllegalStateException("Unknown step indicator "
811                         + step + " from RouteDirector.");
812             }
813 
814         } while (step > HttpRouteDirector.COMPLETE);
815

819     /**
820      * Creates a tunnel to the target server.
821      * The connection must be established to the (last) proxy.
822      * A CONNECT request for tunnelling through the proxy will
823      * be created and sent, the response received and checked.
824      * This method does <i>not</i> update the connection with
825      * information about the tunnel, that is left to the caller.
826      *
827      * @param route     the route to establish
828      * @param context   the context for request execution
829      *
830      * @return  <code>true</code> if the tunnelled route is secure,
831      *          <code>false</code> otherwise.
832      *          The implementation here always returns <code>false</code>,
833      *          but derived classes may override.
834      *
835      * @throws HttpException    in case of a problem
836      * @throws IOException      in case of an IO problem
837      */
838     protected boolean createTunnelToTarget(final HttpRoute route,
839                                            final HttpContext context)
840         throws HttpException, IOException {
841 
842         final HttpHost proxy = route.getProxyHost();
843         final HttpHost target = route.getTargetHost();
844         HttpResponse response = null;
845 
846         for (;;) {
847             if (!this.managedConn.isOpen()) {
848                 this.managedConn.open(route, context, this.params);
849             }
850 
851             final HttpRequest connect = createConnectRequest(route, context);
852             connect.setParams(this.params);
853 
854             // Populate the execution context
855             context.setAttribute(ExecutionContext.HTTP_TARGET_HOST, target);
856             context.setAttribute(ExecutionContext.HTTP_PROXY_HOST, proxy);
857             context.setAttribute(ExecutionContext.HTTP_CONNECTION, managedConn);
858             context.setAttribute(ExecutionContext.HTTP_REQUEST, connect);
859 
860             this.requestExec.preProcess(connect, this.httpProcessor, context);
861 
862             response = this.requestExec.execute(connect, this.managedConn, context);
863 
864             response.setParams(this.params);
865             this.requestExec.postProcess(response, this.httpProcessor, context);
866 
867             final int status = response.getStatusLine().getStatusCode();
868             if (status < 200) {
869                 throw new HttpException("Unexpected response to CONNECT request: " +
870                         response.getStatusLine());
871             }
872 
873             if (HttpClientParams.isAuthenticating(this.params)) {
874                 if (this.authenticator.isAuthenticationRequested(proxy, response,
875                         this.proxyAuthStrategy, this.proxyAuthState, context)) {
876                     if (this.authenticator.authenticate(proxy, response,
877                             this.proxyAuthStrategy, this.proxyAuthState, context)) {
878                         // Retry request
879                         if (this.reuseStrategy.keepAlive(response, context)) {
880                             this.log.debug("Connection kept alive");
881                             // Consume response content
882                             final HttpEntity entity = response.getEntity();
883                             EntityUtils.consume(entity);
884                         } else {
885                             this.managedConn.close();
886                         }
887                     } else {
888                         break;
889                     }
890                 } else {
891                     break;
892                 }
893             }
894         }
895 
896         final int status = response.getStatusLine().getStatusCode();
897 
898         if (status > 299) {
899 
900             // Buffer response content
901             final HttpEntity entity = response.getEntity();
902             if (entity != null) {
903                 response.setEntity(new BufferedHttpEntity(entity));
904             }
905 
906             this.managedConn.close();
907             throw new TunnelRefusedException("CONNECT refused by proxy: " +
908                     response.getStatusLine(), response);
909         }
910 
911         this.managedConn.markReusable();
912 
913         // How to decide on security of the tunnelled connection?
914         // The socket factory knows only about the segment to the proxy.
915         // Even if that is secure, the hop to the target may be insecure.
916         // Leave it to derived classes, consider insecure by default here.
917         return false;
918 
919     } // createTunnelToTarget

959      * Creates the CONNECT request for tunnelling.
960      * Called by {@link #createTunnelToTarget createTunnelToTarget}.
961      *
962      * @param route     the route to establish
963      * @param context   the context for request execution
964      *
965      * @return  the CONNECT request for tunnelling
966      */
967     protected HttpRequest createConnectRequest(final HttpRoute route,
968                                                final HttpContext context) {
969         // see RFC 2817, section 5.2 and
970         // INTERNET-DRAFT: Tunneling TCP based protocols through
971         // Web proxy servers
972 
973         final HttpHost target = route.getTargetHost();
974 
975         final String host = target.getHostName();
976         int port = target.getPort();
977         if (port < 0) {
978             final Scheme scheme = connManager.getSchemeRegistry().
979                 getScheme(target.getSchemeName());
980             port = scheme.getDefaultPort();
981         }
982 
983         final StringBuilder buffer = new StringBuilder(host.length() + 6);
984         buffer.append(host);
985         buffer.append(':');
986         buffer.append(Integer.toString(port));
987 
988         final String authority = buffer.toString();
989         final ProtocolVersion ver = HttpProtocolParams.getVersion(params);
990         final HttpRequest req = new BasicHttpRequest
991             ("CONNECT", authority, ver);
992 
993         return req;

997     /**
998      * Analyzes a response to check need for a followup.
999      *
1000      * @param roureq    the request and route.
1001      * @param response  the response to analayze
1002      * @param context   the context used for the current request execution
1003      *
1004      * @return  the followup request and route if there is a followup, or
1005      *          <code>null</code> if the response should be returned as is
1006      *
1007      * @throws HttpException    in case of a problem
1008      * @throws IOException      in case of an IO problem
1009      */
1010     protected RoutedRequest handleResponse(final RoutedRequest roureq,
1011                                            final HttpResponse response,
1012                                            final HttpContext context)
1013         throws HttpException, IOException {
1014 
1015         final HttpRoute route = roureq.getRoute();
1016         final RequestWrapper request = roureq.getRequest();
1017 
1018         final HttpParams params = request.getParams();
1019 
1020         if (HttpClientParams.isAuthenticating(params)) {
1021             HttpHost target = (HttpHost) context.getAttribute(ExecutionContext.HTTP_TARGET_HOST);
1022             if (target == null) {
1023                 target = route.getTargetHost();
1024             }
1025             if (target.getPort() < 0) {
1026                 final Scheme scheme = connManager.getSchemeRegistry().getScheme(target);
1027                 target = new HttpHost(target.getHostName(), scheme.getDefaultPort(), target.getSchemeName());
1028             }
1029 
1030             final boolean targetAuthRequested = this.authenticator.isAuthenticationRequested(
1031                     target, response, this.targetAuthStrategy, targetAuthState, context);
1032 
1033             HttpHost proxy = route.getProxyHost();
1034             // if proxy is not set use target host instead
1035             if (proxy == null) {
1036                 proxy = route.getTargetHost();
1037             }
1038             final boolean proxyAuthRequested = this.authenticator.isAuthenticationRequested(
1039                     proxy, response, this.proxyAuthStrategy, proxyAuthState, context);
1040 
1041             if (targetAuthRequested) {
1042                 if (this.authenticator.authenticate(target, response,
1043                         this.targetAuthStrategy, this.targetAuthState, context)) {
1044                     // Re-try the same request via the same route
1045                     return roureq;
1046                 }
1047             }
1048             if (proxyAuthRequested) {
1049                 if (this.authenticator.authenticate(proxy, response,
1050                         this.proxyAuthStrategy, this.proxyAuthState, context)) {
1051                     // Re-try the same request via the same route
1052                     return roureq;
1053                 }
1054             }
1055         }
1056 
1057         if (HttpClientParams.isRedirecting(params) &&
1058                 this.redirectStrategy.isRedirected(request, response, context)) {
1059 
1060             if (redirectCount >= maxRedirects) {
1061                 throw new RedirectException("Maximum redirects ("
1062                         + maxRedirects + ") exceeded");
1063             }
1064             redirectCount++;
1065 
1066             // Virtual host cannot be used any longer
1067             virtualHost = null;
1068 
1069             final HttpUriRequest redirect = redirectStrategy.getRedirect(request, response, context);
1070             final HttpRequest orig = request.getOriginal();
1071             redirect.setHeaders(orig.getAllHeaders());
1072 
1073             final URI uri = redirect.getURI();
1074             final HttpHost newTarget = URIUtils.extractHost(uri);
1075             if (newTarget == null) {
1076                 throw new ProtocolException("Redirect URI does not specify a valid host name: " + uri);
1077             }
1078 
1079             // Reset auth states if redirecting to another host
1080             if (!route.getTargetHost().equals(newTarget)) {
1081                 this.log.debug("Resetting target auth state");
1082                 targetAuthState.reset();
1083                 final AuthScheme authScheme = proxyAuthState.getAuthScheme();
1084                 if (authScheme != null && authScheme.isConnectionBased()) {
1085                     this.log.debug("Resetting proxy auth state");
1086                     proxyAuthState.reset();
1087                 }
1088             }
1089 
1090             final RequestWrapper wrapper = wrapRequest(redirect);
1091             wrapper.setParams(params);
1092 
1093             final HttpRoute newRoute = determineRoute(newTarget, wrapper, context);
1094             final RoutedRequest newRequest = new RoutedRequest(wrapper, newRoute);
1095 
1096             if (this.log.isDebugEnabled()) {
1097                 this.log.debug("Redirecting to '" + uri + "' via " + newRoute);
1098             }
1099 
1100             return newRequest;
1101         }
1102 
1103         return null;
