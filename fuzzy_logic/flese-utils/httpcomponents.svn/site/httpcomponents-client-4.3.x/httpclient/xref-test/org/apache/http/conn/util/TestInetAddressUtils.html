28  package org.apache.http.conn.util;
29  
30  import org.junit.Assert;
31  import org.junit.Test;
32  
33  /**
34   * Unit tests for InetAddressUtils.
35   */
36  public class TestInetAddressUtils {
37  
38      @Test
39      public void testValidIPv4Address() {
40          Assert.assertTrue(InetAddressUtils.isIPv4Address("127.0.0.1"));
41          Assert.assertTrue(InetAddressUtils.isIPv4Address("192.168.0.0"));
42          Assert.assertTrue(InetAddressUtils.isIPv4Address("255.255.255.255"));
43      }
44  
45      @Test
46      public void testInvalidIPv4Address() {
47          Assert.assertFalse(InetAddressUtils.isIPv4Address(" 127.0.0.1 "));  // Blanks not allowed
48          Assert.assertFalse(InetAddressUtils.isIPv4Address("g.ar.ba.ge"));
49          Assert.assertFalse(InetAddressUtils.isIPv4Address("192.168.0"));
50          Assert.assertFalse(InetAddressUtils.isIPv4Address("256.255.255.255"));
51      }
52  
53      @Test
54      public void testValidIPv6Address() {
55          Assert.assertTrue(InetAddressUtils.isIPv6StdAddress("2001:0db8:0000:0000:0000:0000:1428:57ab"));
56          Assert.assertTrue(InetAddressUtils.isIPv6StdAddress("2001:db8:0:0:0:0:1428:57ab"));
57          Assert.assertTrue(InetAddressUtils.isIPv6StdAddress("0:0:0:0:0:0:0:0"));
58          Assert.assertTrue(InetAddressUtils.isIPv6HexCompressedAddress("2001:0db8:0:0::1428:57ab"));
59          Assert.assertTrue(InetAddressUtils.isIPv6HexCompressedAddress("2001:0db8::1428:57ab"));
60          Assert.assertTrue(InetAddressUtils.isIPv6HexCompressedAddress("2001:db8::1428:57ab"));
61          Assert.assertTrue(InetAddressUtils.isIPv6HexCompressedAddress("::1"));
62          Assert.assertTrue(InetAddressUtils.isIPv6HexCompressedAddress("::")); // http://tools.ietf.org/html/rfc4291#section-2.2
63      }
64  
65      @Test
66      public void testInvalidIPv6Address() {
67          Assert.assertFalse(InetAddressUtils.isIPv6Address("2001:0db8:0000:garb:age0:0000:1428:57ab"));
68          Assert.assertFalse(InetAddressUtils.isIPv6Address("2001:0gb8:0000:0000:0000:0000:1428:57ab"));
69          Assert.assertFalse(InetAddressUtils.isIPv6StdAddress("0:0:0:0:0:0:0:0:0")); // Too many
70          Assert.assertFalse(InetAddressUtils.isIPv6StdAddress("0:0:0:0:0:0:0")); // Too few
71          Assert.assertFalse(InetAddressUtils.isIPv6HexCompressedAddress(":1"));
72          Assert.assertFalse(InetAddressUtils.isIPv6Address("2001:0db8::0000::57ab")); // Cannot have two contractions
73          Assert.assertFalse(InetAddressUtils.isIPv6HexCompressedAddress("1:2:3:4:5:6:7::9")); // too many fields before ::
74          Assert.assertFalse(InetAddressUtils.isIPv6HexCompressedAddress("1::3:4:5:6:7:8:9")); // too many fields after ::
75          Assert.assertFalse(InetAddressUtils.isIPv6HexCompressedAddress("::3:4:5:6:7:8:9")); // too many fields after ::
76      }
77  
78      @Test
79      // Test HTTPCLIENT-1319
80      public void testInvalidIPv6AddressIncorrectGroupCount() {
81          Assert.assertFalse(InetAddressUtils.isIPv6HexCompressedAddress("1:2::4:5:6:7:8:9")); // too many fields in total
82          Assert.assertFalse(InetAddressUtils.isIPv6HexCompressedAddress("1:2:3:4:5:6::8:9")); // too many fields in total
83      }
84  
85      @Test
86      public void testValidIPv4MappedIPv6Address() {
87          Assert.assertTrue(InetAddressUtils.isIPv4MappedIPv64Address("::FFFF:1.2.3.4"));
88          Assert.assertTrue(InetAddressUtils.isIPv4MappedIPv64Address("::ffff:255.255.255.255"));
89      }
90  
91      @Test
92      public void testInValidIPv4MappedIPv6Address() {
93          Assert.assertFalse(InetAddressUtils.isIPv4MappedIPv64Address("2001:0db8:0000:0000:0000:0000:1428:57ab"));
94          Assert.assertFalse(InetAddressUtils.isIPv4MappedIPv64Address("::ffff:1:2:3:4"));
95      }
