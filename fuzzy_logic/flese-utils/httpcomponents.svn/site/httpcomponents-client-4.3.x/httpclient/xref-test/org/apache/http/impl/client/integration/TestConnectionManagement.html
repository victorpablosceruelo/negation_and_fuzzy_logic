101     @Test
102     public void testReleaseConnection() throws Exception {
103 
104         final PoolingHttpClientConnectionManager mgr = new PoolingHttpClientConnectionManager();
105         mgr.setMaxTotal(1);
106 
107         final HttpHost target = getServerHttp();
108         final HttpRoute route = new HttpRoute(target, null, false);
109         final int      rsplen = 8;
110         final String      uri = "/random/" + rsplen;
111 
112         final HttpRequest request = new BasicHttpRequest("GET", uri, HttpVersion.HTTP_1_1);
113         final HttpContext context = new BasicHttpContext();
114 
115         HttpClientConnection conn = getConnection(mgr, route);
116         mgr.connect(conn, route, 0, context);
117         mgr.routeComplete(conn, route, context);
118 
119         context.setAttribute(HttpCoreContext.HTTP_CONNECTION, conn);
120         context.setAttribute(HttpCoreContext.HTTP_TARGET_HOST, target);
121 
122         final HttpProcessor httpProcessor = new ImmutableHttpProcessor(
123                 new HttpRequestInterceptor[] { new RequestContent(), new RequestConnControl() });
124 
125         final HttpRequestExecutor exec = new HttpRequestExecutor();
126         exec.preProcess(request, httpProcessor, context);
127         HttpResponse response = exec.execute(request, conn, context);
128 
129         Assert.assertEquals("wrong status in first response",
130                      HttpStatus.SC_OK,
131                      response.getStatusLine().getStatusCode());
132         byte[] data = EntityUtils.toByteArray(response.getEntity());
133         Assert.assertEquals("wrong length of first response entity",
134                      rsplen, data.length);
135         // ignore data, but it must be read
136 
137         // check that there is no auto-release by default
138         try {
139             // this should fail quickly, connection has not been released
140             getConnection(mgr, route, 10L, TimeUnit.MILLISECONDS);
141             Assert.fail("ConnectionPoolTimeoutException should have been thrown");
142         } catch (final ConnectionPoolTimeoutException e) {
143             // expected
144         }
145 
146         conn.close();
147         mgr.releaseConnection(conn, null, -1, null);
148         conn = getConnection(mgr, route);
149         Assert.assertFalse("connection should have been closed", conn.isOpen());
150 
151         mgr.connect(conn, route, 0, context);
152         mgr.routeComplete(conn, route, context);
153 
154         // repeat the communication, no need to prepare the request again
155         context.setAttribute(HttpCoreContext.HTTP_CONNECTION, conn);
156         response = exec.execute(request, conn, context);
157 
158         Assert.assertEquals("wrong status in second response",
159                      HttpStatus.SC_OK,
160                      response.getStatusLine().getStatusCode());
161         data = EntityUtils.toByteArray(response.getEntity());
162         Assert.assertEquals("wrong length of second response entity",
163                      rsplen, data.length);
164         // ignore data, but it must be read
165 
166         // release connection after marking it for re-use
167         // expect the next connection obtained to be open
168         mgr.releaseConnection(conn, null, -1, null);
169         conn = getConnection(mgr, route);
170         Assert.assertTrue("connection should have been open", conn.isOpen());
171 
172         // repeat the communication, no need to prepare the request again
173         context.setAttribute(HttpCoreContext.HTTP_CONNECTION, conn);
174         response = exec.execute(request, conn, context);
175 
176         Assert.assertEquals("wrong status in third response",
177                      HttpStatus.SC_OK,
178                      response.getStatusLine().getStatusCode());
179         data = EntityUtils.toByteArray(response.getEntity());
180         Assert.assertEquals("wrong length of third response entity",
181                      rsplen, data.length);
182         // ignore data, but it must be read
183 
184         mgr.releaseConnection(conn, null, -1, null);
185         mgr.shutdown();
186     }

295     @Test
296     public void testCloseExpiredIdleConnections() throws Exception {
297 
298         final PoolingHttpClientConnectionManager mgr = new PoolingHttpClientConnectionManager();
299         mgr.setMaxTotal(1);
300 
301         final HttpHost target = getServerHttp();
302         final HttpRoute route = new HttpRoute(target, null, false);
303         final HttpContext context = new BasicHttpContext();
304 
305         final HttpClientConnection conn = getConnection(mgr, route);
306         mgr.connect(conn, route, 0, context);
307         mgr.routeComplete(conn, route, context);
308 
309         Assert.assertEquals(1, mgr.getTotalStats().getLeased());
310         Assert.assertEquals(1, mgr.getStats(route).getLeased());
311 
312         mgr.releaseConnection(conn, null, 100, TimeUnit.MILLISECONDS);
313 
314         // Released, still active.
315         Assert.assertEquals(1, mgr.getTotalStats().getAvailable());
316         Assert.assertEquals(1, mgr.getStats(route).getAvailable());
317 
318         mgr.closeExpiredConnections();
319 
320         // Time has not expired yet.
321         Assert.assertEquals(1, mgr.getTotalStats().getAvailable());
322         Assert.assertEquals(1, mgr.getStats(route).getAvailable());
323 
324         Thread.sleep(150);
325 
326         mgr.closeExpiredConnections();
327 
328         // Time expired now, connections are destroyed.
329         Assert.assertEquals(0, mgr.getTotalStats().getAvailable());
330         Assert.assertEquals(0, mgr.getStats(route).getAvailable());
331 
332         mgr.shutdown();
333     }

478 
479         Assert.assertFalse(conn.isOpen());

610     static class LatchSupport {
611 
612         private final CountDownLatch continueLatch;
613         private final CountDownLatch waitLatch = new CountDownLatch(1);
614         protected final WaitPolicy waitPolicy;
615 
616         LatchSupport(final CountDownLatch continueLatch, final WaitPolicy waitPolicy) {
617             this.continueLatch = continueLatch;
618             this.waitPolicy = waitPolicy;
619         }
620 
621         void waitForState() throws InterruptedException {
622             if(!waitLatch.await(1, TimeUnit.SECONDS)) {
623                 throw new RuntimeException("waited too long");
624             }
625         }
626 
627         void latch() {
628             waitLatch.countDown();
629             try {
630                 if (!continueLatch.await(60, TimeUnit.SECONDS)) {
631                     throw new RuntimeException("waited too long!");
632                 }
633             } catch (final InterruptedException e) {
634                 throw new RuntimeException(e);
635             }
636         }
637     }
638 
639     private static class StallingSocketFactory extends LatchSupport implements ConnectionSocketFactory {
640 
641         private final ConnectionSocketFactory delegate;
642 
643         public StallingSocketFactory(
644                 final CountDownLatch continueLatch,
645                 final WaitPolicy waitPolicy,
646                 final ConnectionSocketFactory delegate) {
647             super(continueLatch, waitPolicy);
648             this.delegate = delegate;
649         }
650 
651         public Socket connectSocket(
652                 final int connectTimeout,
653                 final Socket sock,
654                 final HttpHost host,
655                 final InetSocketAddress remoteAddress,
656                 final InetSocketAddress localAddress,
657                 final HttpContext context) throws IOException, ConnectTimeoutException {
658             if(waitPolicy == WaitPolicy.BEFORE_CONNECT) {
659                 latch();
660             }
661 
662             final Socket socket = delegate.connectSocket(
663                     connectTimeout, sock, host, remoteAddress, localAddress, context);
664 
665             if(waitPolicy == WaitPolicy.AFTER_CONNECT) {
666                 latch();
667             }
668 
669             return socket;
670         }
671 
672         public Socket createSocket(final HttpContext context) throws IOException {
673             if(waitPolicy == WaitPolicy.BEFORE_CREATE) {
674                 latch();
