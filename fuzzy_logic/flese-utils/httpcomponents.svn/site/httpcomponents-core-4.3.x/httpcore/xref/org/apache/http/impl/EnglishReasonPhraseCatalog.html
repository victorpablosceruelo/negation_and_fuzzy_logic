37  /**
38   * English reason phrases for HTTP status codes.
39   * All status codes defined in RFC1945 (HTTP/1.0), RFC2616 (HTTP/1.1), and
40   * RFC2518 (WebDAV) are supported.
41   *
42   * @since 4.0
43   */
44  @Immutable
45  public class EnglishReasonPhraseCatalog implements ReasonPhraseCatalog {
46  
47      // static array with english reason phrases defined below
48  
49      /**
50       * The default instance of this catalog.
51       * This catalog is thread safe, so there typically
52       * is no need to create other instances.
53       */
54      public final static EnglishReasonPhraseCatalog INSTANCE =
55          new EnglishReasonPhraseCatalog();
56  
57  
58      /**
59       * Restricted default constructor, for derived classes.
60       * If you need an instance of this class, use {@link #INSTANCE INSTANCE}.
61       */
62      protected EnglishReasonPhraseCatalog() {
63          // no body
64      }
65  
66  
67      /**
68       * Obtains the reason phrase for a status code.
69       *
70       * @param status    the status code, in the range 100-599
71       * @param loc       ignored
72       *
73       * @return  the reason phrase, or <code>null</code>
74       */
75      public String getReason(final int status, final Locale loc) {
76          Args.check(status >= 100 && status < 600, "Unknown category for status code " + status);
77          final int category = status / 100;
78          final int subcode  = status - 100*category;
79  
80          String reason = null;
81          if (REASON_PHRASES[category].length > subcode) {
82              reason = REASON_PHRASES[category][subcode];
83          }
84  
85          return reason;
86      }
87  
88  
89      /** Reason phrases lookup table. */
90      private static final String[][] REASON_PHRASES = new String[][]{
91          null,
92          new String[3],  // 1xx
93          new String[8],  // 2xx
94          new String[8],  // 3xx
95          new String[25], // 4xx
96          new String[8]   // 5xx
97      };
98  
99  
100 
101     /**
102      * Stores the given reason phrase, by status code.
103      * Helper method to initialize the static lookup table.
104      *
105      * @param status    the status code for which to define the phrase
106      * @param reason    the reason phrase for this status code
107      */
108     private static void setReason(final int status, final String reason) {
109         final int category = status / 100;
110         final int subcode  = status - 100*category;
111         REASON_PHRASES[category][subcode] = reason;
112     }
