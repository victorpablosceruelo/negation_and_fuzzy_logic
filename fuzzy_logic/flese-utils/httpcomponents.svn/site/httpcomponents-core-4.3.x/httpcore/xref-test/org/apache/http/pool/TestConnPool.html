47      static class LocalPoolEntry extends PoolEntry<String, HttpConnection> {
48  
49          private boolean closed;
50  
51          public LocalPoolEntry(final String route, final HttpConnection conn) {
52              super(null, route, conn);
53          }
54  
55          @Override
56          public void close() {
57              if (this.closed) {
58                  return;
59              }
60              this.closed = true;
61              try {
62                  getConnection().close();
63              } catch (final IOException ignore) {
64              }
65          }
66  
67          @Override
68          public boolean isClosed() {
69              return this.closed;
70          }
71  
72      }
73  
74      static class LocalConnPool extends AbstractConnPool<String, HttpConnection, LocalPoolEntry> {
75  
76          public LocalConnPool(
77                  final ConnFactory<String, HttpConnection> connFactory,
78                  final int defaultMaxPerRoute, final int maxTotal) {
79              super(connFactory, defaultMaxPerRoute, maxTotal);
80          }
81  
82          @Override
83          protected LocalPoolEntry createEntry(final String route, final HttpConnection conn) {
84              return new LocalPoolEntry(route, conn);
85          }
86  
87      }

179     static class GetPoolEntryThread extends Thread {
180 
181         private final Future<LocalPoolEntry> future;
182         private final long time;
183         private final TimeUnit tunit;
184 
185         private volatile LocalPoolEntry entry;
186         private volatile Exception ex;
187 
188         GetPoolEntryThread(final Future<LocalPoolEntry> future, final long time, final TimeUnit tunit) {
189             super();
190             this.future = future;
191             this.time = time;
192             this.tunit = tunit;
193             setDaemon(true);
194         }
195 
196         GetPoolEntryThread(final Future<LocalPoolEntry> future) {
197             this(future, 1000, TimeUnit.SECONDS);
198         }
199 
200         @Override
201         public void run() {
202             try {
203                 this.entry = this.future.get(this.time, this.tunit);
204             } catch (final Exception ex) {
205                 this.ex = ex;
206             }
207         }
208 
209         public boolean isDone() {
210             return this.future.isDone();
211         }
212 
213         public LocalPoolEntry getEntry() {
214             return this.entry;
215         }
216 
217         public Exception getException() {
218             return this.ex;
219         }
220 
221     }
